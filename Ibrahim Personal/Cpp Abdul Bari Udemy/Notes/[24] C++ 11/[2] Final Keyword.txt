- Final keyword restricts Inheritence and function over-riding

Eg. 1:

#include <iostream>

using namespace std;
class Parent final
{

};

class Child: public Parent // Inheritence from the parent class is not allowed here as the parent is marked final
{

};

int main()
{
    cout << "Hello world!" << endl;
    return 0;
}

Eg. 2:

#include <iostream>

using namespace std;
class Parent
{
    virtual void show() final
    {

    }

};

class Child: public Parent
{
    void show() // function over-riding not allowed if the function in the base class is final
    {

    }

};

int main()
{
    cout << "Hello world!" << endl;
    return 0;
}
