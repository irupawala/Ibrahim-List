String solution(String source){
int len = source.length();
Character rightch = source.charAt(0);
int maxcount = 1, count = 1;
       
for(int i = 1; i <len; i++){
  if(source.charAt(i) == source.charAt(i - 1)){
   count ++;
  }
  else{
   if(maxcount < count){
    maxcount = count;
    rightch = source.charAt(i - 1);
   }
   else if(maxcount == count){
    rightch = source.charAt(i - 1);
   }
   count = 1;
  }
}
if(maxcount < count){
   maxcount = count;
   rightch = source.charAt(len - 1);
}
else if(maxcount == count){
  rightch = source.charAt(len - 1);
}
String result = new String(rightch);
result = result + Integer.toString(maxcount);
return result;
}
