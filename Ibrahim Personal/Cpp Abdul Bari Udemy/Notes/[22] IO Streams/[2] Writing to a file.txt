- We have to use fstream for files 
- If the file doesn't exist then new file will be created
- The contents will be truncated if the file already exists 
- If we want to append instead of truncating use 
	ofstream outfile("Mytxt", ios::app)
- If we want to truncate then 
	ofstream outfile("Mytxt", ios::trunc)
- By Default it is truncate, ("Mytxt", ios::trunc)


- It is necessary to close the file, because when program is not running resources should be released.


#include <iostream>
#include <fstream>

using namespace std;

int main()
{
    //cout << "Hello world!" << endl;
    ofstream outfile("My.txt");
    outfile << "Hello" << endl;
    outfile << 25 << endl;
    outfile.close();
    return 0;
}
